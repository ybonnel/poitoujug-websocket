Second test

On doit pouvoir parler tout seul

POJO

public class Message {
    public String user;
    public String text;
    public Date date;

    public Message(String user, String text) {
        this.user = user;
        this.text = text;
        this.date = new Date();
    }
}

Test

private Session openSession(String name, Consumer<Message> c) {
    return client.connect(new WebSocketAdapter() {
        @Override
        public void onWebSocketText(String message) {
            c.accept(ContentType.GSON.fromJson(message, Message.class));
        }
    },new URI("ws://localhost:" + port + "/chat/" + name)).get();
}
@Test
public void canTalkWithMyself() throws Exception {
    List<Message> messages = new ArrayList<>();
    Session session = openSession("user", messages::add);

    assertThat(session).isNotNull();
    assertThat(session.isOpen()).isTrue();

    session.getRemote().sendString("\"i'm alone\"");

    await().until(messages::size, equalTo(1));
    assertThat(messages.get(0)).isEqualToIgnoringGivenFields(
        new Message("user", "i'm alone"), "date");

    session.close();
    await().until(session::isOpen, is(false));
}

Implémentation


public static void routes() {
    websocket("/chat/:name", Main::buildListenner);
}

private static WebSocketListener<String, Message> buildListenner(
        RouteParameters routeParameters) {
    return WebSocketListener.<String, Message>newBuilder(String.class)
            .onMessage((session, message) -> {
                try {
                    session.sendMessage(new Message(
                        routeParameters.getParam("name"), message));
                } catch (IOException ignore) {
                }
            }).build();
}

Implémentation

static Set<WebSocketSession<Message>> sessionOpened = new HashSet<>();

private static WebSocketListener<String, Message> buildListenner(
        RouteParameters routeParameters) {
    return WebSocketListener.<String, Message>newBuilder(String.class)
            .onConnect(sessionOpened::add)
            .onClose(sessionOpened::remove)
            .onMessage(text -> sendMessageToAllSession(new Message(
                    routeParameters.getParam("name"), text))
            ).build();
}

private static void sendMessageToAllSession(Message message) {
    sessionOpened.forEach(session -> {
        try {
            session.sendMessage(message);
        } catch (IOException ignore) {}
    });
}

Code javascript

app.controller('ChatCtrl', function ($scope, $timeout) {
    $scope.loggedIn = false;
    $scope.messages = [];
    $scope.login = function(userTmp) {
        $scope.user = userTmp;
        $scope.ws = new WebSocket("ws://localhost:9999/chat/" + $scope.user);
        $scope.ws.onmessage = function (message) {
            $scope.$apply(function () {
                $scope.messages.unshift(JSON.parse(message.data));
            });
        };
        $scope.ws.onopen = function () {
            $scope.ws.send(JSON.stringify("connected"));
        };
        $scope.loggedIn = true;
    };

    $scope.logout = function() {
        $scope.ws.send(JSON.stringify("disconnected"));
        $timeout(function () {
            $scope.ws.close();
            $scope.loggedIn = false;
        })
    };

    $scope.sendMessage = function() {
        $scope.ws.send(JSON.stringify($scope.message));
        $scope.message = '';
    }
});
